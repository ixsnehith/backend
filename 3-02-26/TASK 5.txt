TASK 5 
— COUNT 
1.SELECT COUNT(*) AS total_students
FROM STUDENTS;
.SELECT COUNT(*) AS active_students
FROM STUDENTS
WHERE active_flag = 1;

 — SUM / AVG 
1.SELECT SUM(fees) AS total_course_fees
FROM COURSES;
.SELECT AVG(fees) AS average_course_fee
FROM COURSES;
.SELECT SUM(fees) AS sum_high_fee_courses
FROM COURSES
WHERE fees > 2000;

— MIN / MAX 
1.SELECT MIN(age) AS min_student_age
FROM STUDENTS;
.SELECT MAX(fees) AS max_course_fee
FROM COURSES;
.SELECT MAX(registration_date) AS latest_join_date
FROM STUDENTS;
.SELECT MIN(registration_date) AS oldest_join_date
FROM STUDENTS;

— GROUP BY 
1.SELECT active_flag, COUNT(*) AS student_count
FROM STUDENTS
GROUP BY active_flag;
.SELECT age, COUNT(*) AS students_per_age
FROM STUDENTS
GROUP BY age;
.SELECT course_name, SUM(fees) AS total_fees
FROM COURSES
GROUP BY course_name;

 — HAVING 
1.SELECT course_id, COUNT(*) AS enrollment_count
FROM ENROLLMENTS
GROUP BY course_id
HAVING COUNT(*) > 3;
.SELECT age, AVG(age) AS avg_age
FROM STUDENTS
GROUP BY age
HAVING AVG(age) > 21;
.SELECT course_id, COUNT(*) AS total_enrollments
FROM ENROLLMENTS
GROUP BY course_id
HAVING COUNT(*) > 2;
